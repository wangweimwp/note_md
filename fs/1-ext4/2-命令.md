```bash
#!/bin/bash

#制作一个有htree目录结构的ext4镜像，只有目录足够多时候才会启用htree
# 1. 创建带哈希树的目录（已存在可跳过）
dd if=/dev/zero of=ext4_htree.img bs=1M count=100
mkfs.ext4 -O dir_index ext4_htree.img
mkdir -p /mnt/ext4_htree
mount -o loop ext4_htree.img /mnt/ext4_htree
mkdir /mnt/ext4_htree/large_dir

# 创建足够多的文件触发哈希树（至少2000+）
for i in {1..5000}; do
  touch /mnt/ext4_htree/large_dir/file_$i
done
umount /mnt/ext4_htree

# 2. 提取根节点信息
debugfs -R "htree /large_dir" ext4_htree.img | awk '/Root node dump/{flag=1} flag&&/^$/{exit} flag'
```
